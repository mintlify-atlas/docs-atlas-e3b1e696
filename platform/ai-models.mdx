    ```typescript convex/llm.ts:16-45
    async function fetchWithRetry(
      url: string,
      init: RequestInit,
      maxRetries = 3
    ): Promise<Response> {
      let lastError: Error | null = null;
      
      for (let attempt = 0; attempt <= maxRetries; attempt++) {
        try {
          const response = await fetch(url, init);
          
          // Retry on transient errors
          if ((response.status === 429 || // Rate limit
               response.status === 502 || // Bad gateway
               response.status === 503) && // Service unavailable
              attempt < maxRetries) {
            const delay = Math.min(1000 * Math.pow(2, attempt), 8000);
            await new Promise(resolve => setTimeout(resolve, delay));
            continue;
          }
          
          return response;
        } catch (err) {
          lastError = err instanceof Error ? err : new Error(String(err));
          if (attempt < maxRetries) {
            const delay = Math.min(1000 * Math.pow(2, attempt), 8000);
            await new Promise(resolve => setTimeout(resolve, delay));
          }
        }
      }
      
      throw lastError ?? new Error("Request failed after retries");
    }
    ```

### Batch Embedding (Used in Pipeline)

```typescript convex/llm.ts:77-107
export const embedTextBatch = action({
  args: { texts: v.array(v.string()) },
  handler: async (_ctx, args) => {
    if (args.texts.length === 0) return [];
    
    const response = await fetchWithRetry(
      "https://api.openai.com/v1/embeddings",
      {
        method: "POST",
        headers: {
          Authorization: `Bearer ${OPENAI_API_KEY}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: OPENAI_EMBED_MODEL,
          input: args.texts, // Array of strings
        }),
      }
    );
    
    const data = await response.json();
    
    // OpenAI returns data with index field - sort to maintain order
    const sorted = data.data.sort((a, b) => a.index - b.index);
    return sorted.map(d => d.embedding); // number[][]
  },
});
```

<Note>
**Why batch?** OpenAI's embeddings API accepts up to 2048 inputs in a single request. Batching 20 texts reduces:
- Network round trips (20x faster)
- API call costs (flat fee per request)
- Rate limit pressure
</Note>

<Accordion title="Why store embeddings in Convex vs vector DB?">
  **Pros:**
  - Single data store (no Pinecone/Weaviate needed)
  - Cosine similarity computed in-memory (fast for under 10k vectors)
  - Simpler deployment and debugging
  - Convex handles persistence and indexes
  
  **Cons:**
  - Not optimized for >100k vectors (HNSW, ANN)
  - No built-in vector indexing
  - Loads all vectors into memory for search
  
  **Current scale:** 100-500 chunks per person. In-memory cosine similarity is sufficient.
  
  **Migration path:** If scale exceeds 10k vectors/query, migrate to Pinecone:

## Semantic Search (RAG)

### Cosine Similarity

```typescript convex/pipeline.ts:140-152
function cosineSimilarity(a: number[], b: number[]): number {
  if (a.length === 0 || b.length === 0 || a.length !== b.length) return 0;
  
  let dot = 0;
  let magA = 0;
  let magB = 0;
  
  for (let i = 0; i < a.length; i++) {
    dot += a[i] * b[i];
    magA += a[i] * a[i];
    magB += b[i] * b[i];
  }
  
  if (!magA || !magB) return 0;
  return dot / (Math.sqrt(magA) * Math.sqrt(magB));
}
```

<Tabs>
  <Tab title="What is Cosine Similarity?">
    Measures the angle between two vectors, ranging from -1 to 1:
    - **1.0** = identical direction (very similar)
    - **0.0** = orthogonal (unrelated)
    - **-1.0** = opposite direction (contradictory)
    
    ```
    Vector A: [0.2, 0.8, 0.1]
    Vector B: [0.3, 0.7, 0.2]
    
    dot(A,B) = 0.2*0.3 + 0.8*0.7 + 0.1*0.2 = 0.64
    |A| = sqrt(0.04 + 0.64 + 0.01) = 0.83
    |B| = sqrt(0.09 + 0.49 + 0.04) = 0.79
    
    cos_sim = 0.64 / (0.83 * 0.79) = 0.976 (highly similar)
    ```
  </Tab>
  
  <Tab title="Why Cosine vs Euclidean?">
    **Cosine similarity** measures **direction**, not magnitude:
    - Better for text embeddings (captures semantic meaning)
    - Normalized by vector length (fair comparison)
    - Standard in NLP and information retrieval
    
    **Euclidean distance** measures **magnitude**:
    - Sensitive to vector length
    - Less robust for high-dimensional data
    - Used in clustering, image recognition
  </Tab>
</Tabs>

### Retrieval Function

```typescript convex/pipeline.ts:1406-1451
export const scoreAndSelectChunks = action({
  args: {
    personId: v.id("persons"),
    stageId: v.id("stages"),
    query: v.string(),
    topK: v.optional(v.number()),
  },
  handler: async (ctx, args) => {
    const topK = args.topK ?? 8;
    
    // Step 1: Embed the query
    const qEmbedding: number[] = await ctx.runAction(
      api.llm.embedText,
      { text: args.query }
    );
    
    // Step 2: Get all chunks for this stage
    const stageChunks = await ctx.runQuery(
      internal.pipeline.listChunksByStage,
      { stageId: args.stageId }
    );
    
    // Step 3: Score and rank
    const scoredStage = stageChunks
      .map(chunk => ({
        chunk,
        score: cosineSimilarity(qEmbedding, chunk.embedding)
      }))
      .sort((a, b) => b.score - a.score)
      .slice(0, topK);
    
    let usedFallback = false;
    let result = scoredStage;
    
    // Step 4: Fallback to all chunks if stage-scoped results are sparse
    if (scoredStage.length < Math.max(3, Math.floor(topK / 2))) {
      usedFallback = true;
      const allChunks = await ctx.runQuery(
        internal.pipeline.listChunksByPerson,
        { personId: args.personId }
      );
      result = allChunks
        .map(chunk => ({
          chunk,
          score: cosineSimilarity(qEmbedding, chunk.embedding)
        }))
        .sort((a, b) => b.score - a.score)
        .slice(0, topK);
    }
    
    return {
      usedFallback,
      chunks: result.map(item => ({
        id: item.chunk._id,
        text: item.chunk.text,
        citation: item.chunk.citation,
        score: item.score,
      })),
    };
  },
});
```

<Steps>
  <Step title="Embed query">
    User question → OpenAI embeddings → 1536-dim vector
  </Step>
  
  <Step title="Retrieve stage chunks">
    Load all chunks linked to the current stage (typically 30-100 chunks)
  </Step>
  
  <Step title="Compute similarity">
    Score each chunk against query embedding using cosine similarity
  </Step>
  
  <Step title="Rank & select top K">
    Sort by score descending, return top 8 chunks
  </Step>
  
  <Step title="Fallback if sparse">
    If stage has fewer than 3 good matches, expand search to all chunks for this person
  </Step>
</Steps>

<Warning>
**Fallback logic** is critical for:
- New persons with incomplete stage mappings
- Questions spanning multiple eras
- Under-represented stages with few sources

Fallback is indicated in the UI with a subtle badge.
</Warning>

### Voice Extraction

The chat system extracts direct quotes to match the person's speaking style:

```typescript convex/persons.ts:417-448
const voiceSamples: string[] = [];

for (const chunk of retrieval.chunks.slice(0, 5)) {
  const text = chunk.text;
  
  // Extract quoted speech
  const quoteMatches = text.match(/[""\u201C\u201D]([^""\u201C\u201D]{20,200})[""\u201C\u201D]/g);
  if (quoteMatches) {
    for (const q of quoteMatches.slice(0, 2)) {
      voiceSamples.push(q.replace(/[""\u201C\u201D]/g, ""));
    }
  }
  
  // Extract first-person sentences from transcripts
  const sentences = text.split(/[.!?]+/).filter(s => {
    const trimmed = s.trim().toLowerCase();
    return (
      (trimmed.startsWith("i ") || 
       trimmed.startsWith("we ") || 
       trimmed.startsWith("my ")) &&
      trimmed.length > 30 &&
      trimmed.length < 250
    );
  });
  
  for (const s of sentences.slice(0, 2)) {
    voiceSamples.push(s.trim());
  }
}

const uniqueVoice = [...new Set(voiceSamples)].slice(0, 6);
```

### Response Generation

```typescript convex/llm.ts:147-181
export const generateChatReply = action({
  args: {
    system: v.string(),
    user: v.string(),
  },
  handler: async (_ctx, args) => {
    const response = await fetchWithRetry(
      "https://api.openai.com/v1/chat/completions",

        method: "POST",
        headers: {
          Authorization: `Bearer ${OPENAI_API_KEY}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: OPENAI_MODEL, // gpt-5.2
          temperature: 0.3, // Low temp for consistency
          messages: [
            { role: "system", content: args.system },
            { role: "user", content: args.user },
          ],
        }),
      }
    );
    
    if (!response.ok) {
      const errorText = await response.text();
      throw new Error(`Chat completion failed: ${errorText}`);
    }

      "I do not have enough evidence to answer that yet."
    );
  },
});
```

<Tabs>
  <Tab title="Temperature = 0.3">
    **Why low temperature?**
    
    Temperature controls randomness:
    - **0.0** = deterministic, repetitive
    - **0.3** = consistent but natural variation
    - **0.7** = creative but less factual
    - **1.0** = highly random, unpredictable
    
    For biography chat, we want:
    - Consistent voice across messages
    - Grounded in evidence (less creativity)
    - Natural variation in phrasing
    
    **0.3 is the sweet spot** for persona consistency.
  </Tab>
  
  <Tab title="Why GPT-5.2?">
    **Model selection rationale:**
    
    | Model | Pros | Cons |
    |-------|------|------|
    | GPT-4o | Fast, cheap | Less nuanced personas |
    | GPT-4 Turbo | Good balance | Slightly outdated |
    | **GPT-5.2** | Best instruction following, persona accuracy | Higher cost, slower |
    | GPT-3.5 | Very cheap | Poor at persona consistency |
    
    GPT-5.2 excels at:
    - Following complex system prompts
    - Maintaining consistent voice
    - Grounding in provided evidence
    - Avoiding markdown when instructed
  </Tab>

```typescript convex/llm.ts:110-144
export const generateStructuredJson = action({
  args: {
    system: v.string(),
    user: v.string(),
  },
  handler: async (_ctx, args) => {

          model: OPENAI_MODEL,
          temperature: 0.2, // Even lower for structured output
          response_format: { type: "json_object" }, // Force JSON

            { role: "system", content: args.system },
            { role: "user", content: args.user },
          ],
        }),
      }

    if (!response.ok) {

});
```

<Note>
**`response_format: { type: "json_object" }`** guarantees valid JSON output. The LLM will always return parseable JSON, even if the content is `{"error": "..."}`.

This is critical for pipeline reliability - no try/catch needed for JSON parsing.
</Note>

## Cost Optimization

### Token Usage Breakdown

| Operation | Tokens/Call | Calls/Person | Total Tokens | Cost (GPT-5.2 @ $15/M) |
|-----------|-------------|--------------|--------------|------------------------|
| Query generation | ~800 | 1 | 800 | $0.01 |
| Source vetting | ~1,200 | 2-3 | 3,600 | $0.05 |
| Stage generation | ~5,000 | 1 | 5,000 | $0.08 |
| Source mapping | ~1,500 | 20-30 | 40,000 | $0.60 |
| Deep research queries | ~1,000 | 5-7 | 6,000 | $0.09 |
| **Pipeline total** | | | **~55k** | **~$0.83** |
| | | | | |
| Embeddings | ~200 tokens × 300 chunks | 15 batches | 60k tokens | $0.001 (embed model) |
| Chat (per message) | ~2,000-4,000 | Per user | 3,000 avg | $0.05 |

<Warning>
**Highest cost:** Source mapping (30 LLM calls × $0.02 each = $0.60)

**Optimization:** Batch multiple sources per LLM call to reduce this cost by 50%.

**Current implementation:** One source at a time for simplicity, but batching 3-5 sources would reduce costs significantly.
</Warning>

| Metric | Target | Actual | Notes |
|--------|--------|--------|-------|
| Embedding latency | Under 500ms | ~300ms | Batch of 20 texts |

<Note>
**RAG precision@8** measures how often the top 8 retrieved chunks are relevant to the user's question. Our system achieves ~82% precision due to:
- Stage-scoped retrieval (fewer irrelevant chunks)
- Fallback mechanism (ensures sufficient context)
- High-quality embeddings (`text-embedding-3-small`)
</Note>
